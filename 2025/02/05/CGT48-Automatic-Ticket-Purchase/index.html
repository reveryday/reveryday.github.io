Initial

（需手动登入 仅限购一张）

运行前需配置：

  1. python环境：安装anaconda jupyter notebook

  2. Selenium环境：chrome打开jupyter notebook出现了一些问题，所以用了edge，需安装好edge对应版本的edgedriver，用指令pip install selenium安装好selenium库

用edge打开jupyter notebook，新建一个.ipynb文件，运行以下代码即可

#CII首演第一场抢票脚本
import time
from bs4 import BeautifulSoup
from selenium import webdriver
from selenium.webdriver.edge.service import Service
from selenium.webdriver.common.by import By

driver_path = 'D:\\app data\\edgedriver\\edgedriver_win64\\msedgedriver.exe'
service = Service(executable_path=driver_path)
browser = webdriver.Edge(service=service)

target_url="https://48.gnz48.com/tickets/item/6774?seat_type=3"  
#进入抢票页面
browser.get(target_url)
html_k = browser.execute_script('return document.documentElement.outerHTML')  # 获取当前页面的HTML
bs_k = BeautifulSoup(html_k, 'html.parser') 

# seattype2：VIP座票；seattype3：普通座票；seattype4：普通站票
# 选择座位类型
theseattype = browser.find_element(By.XPATH, '//em[@id="seattype3"]')
browser.execute_script('arguments[0].click();', theseattype)

#定时购买
click_time = 0
while True:  # 无限循环
    if click_time > 0:
        break
    if time.strftime("%Y-%m-%d %H:%M:%S",time.localtime()) >= '2025-02-05 20:30:00':  # 设置抢票时间
        try:
            WebDriverWait(browser, 5).until(
            EC.text_to_be_present_in_element((By.ID, 'buy'), '立即购买')
            )
            determine = WebDriverWait(browser, 5).until(
            EC.element_to_be_clickable((By.ID, 'buy'))
            )
            browser.execute_script('arguments[0].click();', determine)
            click_time += 1
        except Exception as e:
            print(f"未找到购买按钮，继续尝试: {e}")

driver_path = 'D:\\app data\\edgedriver\\edgedriver_win64\\msedgedriver.exe'

max_retries = 20  # 最大重试次数
retry_interval = 0.03 # 重试间隔（秒）
prepare_seconds = 30 # 提前准备时间（秒）
Edge_options = EdgeOptions()

service = Service(executable_path=driver_path)
driver = webdriver.Edge(service=service, options=Edge_options)
#driver = webdriver.Edge(service=service)
driver.get(target_url)
def refresh_page():
    driver.refresh()
    logger.info("页面已刷新")
# 自动点击立即购买
def buy(retry_count=0):
    if retry_count >= max_retries:
        logger.error(f"已达到最大重试次数 {max_retries}，抢票失败")
        return False
    
    logger.info(f"正在进行抢票 (尝试 {retry_count+1}/{max_retries})")
    
    try:
        # 定位购买按钮
        buybutton = WebDriverWait(driver, 3, 0.05).until(
            EC.presence_of_element_located((By.ID, 'buy')))
        buybutton_text = buybutton.text
        logger.info("购买按钮状态: " + buybutton_text)
        
        # 检查按钮是否可点击
        if "开售" in buybutton_text:
            logger.info("商品尚未开售，尝试刷新页面...")
            driver.refresh()
            time.sleep(retry_interval)
            return buy(retry_count + 1)
    except Exception as e:
        logger.error(f"找不到购买按钮: {e}")

    try:
        # 使用JavaScript点击按钮，避免可能的覆盖问题
        driver.execute_script("arguments[0].click();", buybutton)
        logger.info("成功点击购买按钮！")
    except Exception as e:
        logger.error(f"点击购买按钮失败: {e}")

    try:
        # 定位提交订单按钮
        submit_button = WebDriverWait(driver, 3, 0.05).until(
            EC.element_to_be_clickable((By.ID, 'buyorder'))
        )    
        # 使用JavaScript点击
        driver.execute_script("arguments[0].click();", submit_button)
        logger.info("成功提交订单！")
        return True
    except Exception as e:
        logger.error(f"提交订单失败: {e}")
        return False
# 使用 NTP 服务器同步时间
def get_network_time():
    try:
        client = ntplib.NTPClient()
        response = client.request('pool.ntp.org', version=3)
        return datetime.datetime.fromtimestamp(response.tx_time)
    except:
        logger.warning("无法获取网络时间，使用本地时间")
        return datetime.datetime.now()
# 使用 sched 模块和网络时间同步
def run_on_time(target_time_str):
    """使用 sched 模块在目标时间准时运行购买函数。"""
    try:
        target_time = datetime.datetime.strptime(target_time_str, "%H:%M:%S").time()
        now_datetime = get_network_time()  # 获取当前时间
        now = now_datetime.time()        
        logger.info(f"当前时间: {now}, 目标时间: {target_time}")
        
        if now < target_time:
            target_datetime = datetime.datetime.combine(now_datetime.date(), target_time)
            time_diff = (target_datetime - now_datetime).total_seconds()
            
            # 如果距离开售时间还有一段时间，先预加载页面
            if time_diff > prepare_seconds:
                logger.info(f"距离开售还有 {time_diff} 秒，先预加载页面")
                
                # 计算时间差
                prepare_time = time_diff - prepare_seconds
                if prepare_time > 0:
                    logger.info(f"等待 {prepare_time} 秒后开始准备抢票")
                    time.sleep(prepare_time)
            
            prepare_datetime = target_datetime - datetime.timedelta(seconds=prepare_seconds)
            prepare_timestamp = time.mktime(prepare_datetime.timetuple())
            
            # 设置定时任务
            s = sched.scheduler(time.time, time.sleep)
            
            # 开售前刷新页面
            s.enterabs(prepare_timestamp, 2, refresh_page, ())
            
            # 准点开始抢票
            target_timestamp = time.mktime(target_datetime.timetuple())
            s.enterabs(target_timestamp, 1, buy, ())
            
            logger.info(f"将在 {target_time} 开始抢票，提前 {prepare_seconds} 秒准备")
            s.run()
        else:
            logger.info("当前时间已经大于目标时间，立即开始购买...")
            buy()
    except ValueError:
        logger.error("目标时间格式错误，请使用 HH:MM:SS 格式。")
    except Exception as e:
        logger.exception("执行 run_on_time 时发生错误：", exc_info=True)
if __name__ == "__main__":
    try:           
        logger.info("=== SNH48抢票脚本启动 ===")
        logger.info(f"目标URL: {target_url}")
        logger.info(f"目标时间: {target_time_str}")
        run_on_time(target_time_str)
    except KeyboardInterrupt:

        logger.info("程序结束，关闭浏览器")
        try:
            driver.quit()
        except:
            pass
