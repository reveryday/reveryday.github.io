2、依据股票代码查询年报并下载为pdf文件

'''
根据股票代码爬取2015-2025年报
'''

import requests
import pandas as pd
import time
import os

def read_stock_codes(file_path='stock_code.csv'):
    """读取股票代码"""
    try:
        if file_path.endswith('.csv'):
            return pd.read_csv(file_path, dtype=str)['symbol'].tolist()
        else:
            with open(file_path, 'r') as f:
                return [line.strip() for line in f]
    except Exception as e:
        print(f"读取股票代码文件出错: {e}")
        return []

def download_annual_report(stock_code, year):
    """下载指定公司指定年份的年报"""
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Accept': '*/*',
        'Referer': 'http://www.cninfo.com.cn/new/commonUrl/pageOfSearch'
    }
    
    # 处理股票代码格式
    if stock_code.startswith('6'):
        stock_code = f"{stock_code}.SH"
    else:
        stock_code = f"{stock_code}.SZ"
    
    url = "http://www.cninfo.com.cn/new/hisAnnouncement/query"
    params = {
        'pageNum': 1,
        'pageSize': 30,
        'column': 'szse',
        'tabName': 'fulltext',
        'plate': '',
        'stock': stock_code,
        'searchkey': f'{year}年年度报告',
        'secid': '',
        'category': 'category_ndbg_szsh',
        'trade': '',
        'seDate': f'{year}-01-01~{year}-12-31'
    }
    
    try:
        response = requests.post(url, headers=headers, data=params)
        data = response.json()
        
        if data.get('announcements') and len(data['announcements']) > 0:
            for announcement in data['announcements']:
                if '年度报告' in announcement['announcementTitle']:
                    download_url = f"http://static.cninfo.com.cn/{announcement['adjunctUrl']}"
                    file_name = f"{stock_code}_{year}_年报.pdf"
                    
                    # 创建公司目录
                    company_dir = os.path.join('annual_reports', stock_code)
                    os.makedirs(company_dir, exist_ok=True)
                    
                    # 检查文件是否已存在
                    file_path = os.path.join(company_dir, file_name)
                    if os.path.exists(file_path):
                        print(f"文件已存在，跳过: {file_name}")
                        return True
                    
                    # 下载文件
                    pdf_response = requests.get(download_url, headers=headers)
                    if pdf_response.status_code == 200:
                        with open(file_path, 'wb') as f:
                            f.write(pdf_response.content)
                        print(f"已下载: {file_name}")
                        return True
                    
        print(f"未找到 {stock_code} {year}年的年报")
        return False
            
    except Exception as e:
        print(f"下载失败 {stock_code} {year}: {e}")
        return False

def main():
    # 创建主目录
    os.makedirs('annual_reports', exist_ok=True)
    
    # 读取股票代码
    stock_codes = read_stock_codes()
    years = range(2015, 2026)
    
    for stock_code in stock_codes:
        print(f"\n处理公司: {stock_code}")
        for year in years:
            # 添加延迟避免被封
            time.sleep(2)
            download_annual_report(stock_code, year)

if __name__ == "__main__":
    main()
